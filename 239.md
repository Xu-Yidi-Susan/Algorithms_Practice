class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        max_nums_list = []
        sub_nums_start = 0
        sub_nums_end = k
        sub_max_index = 0
        
        while sub_nums_end <= len(nums):
            sub_nums = nums[sub_nums_start:sub_nums_end]
            
            if sub_nums_start == 0:
                max_nums = self.find_max(sub_nums)
                max_nums_list.append(max_nums)
                sub_max_index = sub_nums.index(max_nums)
            
            if sub_nums_start > 0:
                if sub_max_index == 0:
                    max_nums = self.find_max(sub_nums)
                    max_nums_list.append(max_nums)
                    sub_max_index = sub_nums.index(max_nums)
                else:
                    if sub_nums[-1] <= max_nums_list[-1]:
                        max_nums_list.append(max_nums_list[-1])
                        sub_max_index -= 1
                    else:
                        max_nums_list.append(sub_nums[-1])
                        sub_max_index = k-1
                                    
            sub_nums_start += 1
            sub_nums_end += 1
            
                
        return max_nums_list  
       
    def find_max(self, sub_nums: List[int]):
        max_num = sub_nums[0]
        for i in range(0, len(sub_nums)):
            if sub_nums[i] > max_num:
                max_num = sub_nums[i]
        return max_num
